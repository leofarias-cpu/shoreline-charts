## Usage

```jsx isStatic
import { Radio, RadioGroup, useRadioState } from '@vtex/admin-ui'

function Example() {
  const state = useRadioState()

  return (
    <RadioGroup state={state} label="Group label">
      <Radio value="radio-1" label="Label" />
      <Radio value="radio-2" label="Label" />
      <Radio value="radio-3" label="Label" />
    </RadioGroup>
  )
}
```

</Section>

## Accessibility

<OrderedList>
  <OrderedListItem>`Radio` has the role `radio`.</OrderedListItem>
  <OrderedListItem>`RadioGroup` has role `radiogroup`.</OrderedListItem>
  <OrderedListItem>
    `Radio` has `aria-checked` set to `true` when it's checked. Otherwise,
    `aria-checked` is set to `false`.
  </OrderedListItem>
  <OrderedListItem>Always use `Radio` within a `RadioGroup`.</OrderedListItem>
