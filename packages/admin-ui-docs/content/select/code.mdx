## Usage

```jsx isStatic
import { useState } from 'react'
import { Select } from '@vtex/admin-ui'

function Example() {
  const [value, setValue] = useState()

  return (
    <Select
      label="Label"
      value={value}
      onChange={(e) => setValue(e.target.value)}
    >
      <option>Option 1</option>
      <option>Option 2</option>
      <option>...</option>
    </Select>
  )
}
```
