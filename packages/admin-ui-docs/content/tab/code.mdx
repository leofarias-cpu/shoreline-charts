## Usage

```jsx isStatic
import {
  Tab,
  TabList,
  TabPanelList,
  TabPanel,
  useTabState,
} from '@vtex/admin-ui'

function Example() {
  const state = useTabState()

  return (
    <>
      <TabList state={state}>
        <Tab>Tab 1</Tab>
        <Tab>Tab 2</Tab>
      </TabList>
      <TabPanelList state={state}>
        <TabPanel>Panel 1</TabPanel>
        <TabPanel>Panel 2</TabPanel>
      </TabPanelList>
    </>
  )
}
```

</Section>
