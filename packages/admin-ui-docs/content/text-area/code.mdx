## Usage

```jsx isStatic
import { useState } from 'react'
import { TextArea } from '@vtex/admin-ui'

function Example() {
  const [value, setValue] = useState()

  return (
    <TextArea
      label="Label"
      value={value}
      onChange={(e) => setValue(e.target.value)}
    />
  )
}
```
