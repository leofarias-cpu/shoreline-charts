## Usage

```jsx
import {
  useCollapse,
  Card,
  CardHeader,
  CardTitle,
  CardContent,
  Box,
} from '@vtex/admin-ui'
function Example = () => {
  const { getToggleProps, getCollapseProps, visible } = useCollapse()
  return (
    <Card>
      <CardHeader
        {...getToggleProps()}
        className={csx({
          cursor: 'pointer',
          ':hover': {
            bg: '$action.neutral.tertiaryHover',
          },
        })}
      >
        <CardInfo>
          <CardTitle>Title</CardTitle>
        </CardInfo>
      </CardHeader>
      <div className={csx({ width: '100%' })} {...getCollapseProps()}>
        <CardContent
          className={csx({
            paddingTop: '$space-6',
          })}
        >
          Text.
        </CardContent>
      </div>
    </Card>
  )
}
```
