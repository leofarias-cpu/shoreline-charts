# SingleSelectFilter

## Usage

```jsx
import {
  experimental_SingleSelectFilter as SingleSelectFilter,
  experimental_useSingleFilterState as useSingleFilterState,
} from '@vtex/admin-ui'

function Example() {
  const state = useSingleFilterState({
    items: [
      { label: 'Available', value: 1, id: '#1' },
      { label: 'Sold out', value: 2, id: '#2' },
    ],
    onChange: () => {},
    label: 'Status',
  })
  return <SingleSelectFilter state={state} />
}
```

## Examples

### Initial values

```jsx noInline live

const SingleSelectFilter = experimental_SingleSelectFilter
const useSingleFilterState = experimental_useSingleFilterState

function Example() {
  const state = useSingleFilterState({
    items: [
      { label: 'Available', value: 1, id: '#1' },
      { label: 'Sold out', value: 2, id: '#2' },
    ],
    onChange: ({ selected }) => console.log(`applied: ${selected}`),
    label: 'Status',
    initialApplied: '#1',
  })
  return <SingleSelectFilter state={state} />
}

render(<Example />)
```

### I18n

The singleselect-filter has default translations. Wrap your app with [I18nProvider](/experimental/localization#i18nprovider/) to configure the locale.

```jsx
import {
  experimental_SingleSelectFilter as SingleSelectFilter,
  experimental_useSingleFilterState as useSingleFilterState,
} from '@vtex/admin-ui'

function Filter() {
  const state = useSingleFilterState({
    items: [
      { label: 'Available', value: 1, id: '#1' },
      { label: 'Sold out', value: 2, id: '#2' },
    ],
    onChange: () => {},
    label: 'Status',
  })
  return <SingleSelectFilter state={state} />
}

function RootApp() {
  return (
    <I18nProvider locale="ko-KO">
      <Filter />
    </I18nProvider>
  )
}

render(<RootApp />)
```
