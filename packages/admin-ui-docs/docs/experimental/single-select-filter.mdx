# SingleSelectFilter

## Usage

```jsx
import {
  experimental_SingleSelectFilter as SingleSelectFilter,
  experimental_useSingleFilterState as useSingleFilterState,
} from '@vtex/admin-ui'

function Example() {
  const state = useSingleFilterState({
    items: [
      { label: 'Available', value: 1, id: '#1' },
      { label: 'Sold out', value: 2, id: '#2' },
    ],
    onChange: () => {},
    label: 'Status',
  })
  return <SingleSelectFilter state={state} />
}
```

## Examples

### Initial values

```jsx noInline live

const SingleSelectFilter = experimental_SingleSelectFilter
const useSingleFilterState = experimental_useSingleFilterState

function Example() {
  const state = useSingleFilterState({
    items: [
      { label: 'Available', value: 1, id: '#1' },
      { label: 'Sold out', value: 2, id: '#2' },
    ],
    onChange: ({ selected }) => console.log(`applied: ${selected}`),
    label: 'Status',
    initialApplied: '#1',
  })
  return <SingleSelectFilter state={state} />
}

render(<Example />)
```
