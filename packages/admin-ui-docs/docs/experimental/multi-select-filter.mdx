# MultiSelectFilter

## Usage

```jsx

import {
  experimental_MultiselectFilter as MultiselectFilter,
  experimental_useMultipleFilterState as useMultipleFilterState,
} from '@vtex/admin-ui'

function Example() {
  const state = useMultipleFilterState({
    items: [
      { label: 'Full', value: 1, id: '#1' },
      { label: 'Empty', value: 2, id: '#2' },
      { label: 'Half empty', value: 4, id: '#3' },
    ],
    onChange: ({ selected }) => console.log(`applied: ${selected}`),
    label: 'Status',
  })
  return <MultiselectFilter state={state} />
}
```

```jsx noInline live
const MultiselectFilter = experimental_MultiselectFilter

function Example() {

    items: [

      { label: 'Unknown', value: 5, id: '#5' },
    ],

    label: 'Status',
    initialApplied: ['#1', '#2'],
  })

  return <MultiselectFilter state={state} />

  
}

render(<Example />)
```
