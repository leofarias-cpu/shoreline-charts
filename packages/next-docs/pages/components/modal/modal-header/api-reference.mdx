# API Reference

Header of the Modal

```jsx
function Example() {
  const [open, setOpen] = React.useState(false);

  return (
    <>
      <Button onClick={() => setOpen(true)}>Open modal</Button>
      <Modal
        open={open}
        onClose={() => {
          setOpen(false);
        }}
      >
        <ModalHeader>
          <ModalHeading>Title</ModalHeading>
        </ModalHeader>
      </Modal>
    </>
  );
}
```
