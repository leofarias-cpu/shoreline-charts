<ComponentSummaryGrid mt="var(--sl-space-4)">
  {allComponentsList.map((component) => {
    console.log(component);
    let href = `/components/${component}`;
    if (component === "checkbox-group") {
      href = `/components/checkbox/${component}`;
    } else if (component === "radio-group") {
      href = `/components/radio/${component}`;
    }
    return <ComponentSummary key={component} name={component} href={href} />;
  })}
</ComponentSummaryGrid>
